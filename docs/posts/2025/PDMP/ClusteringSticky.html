using PlotlyJS

# パラメータ格子
us = collect(-1.0:0.05:1.0)
vs = collect(-1.0:0.05:1.0)
m, n = length(us), length(vs)

# 頂点リスト（flatten した順序で）
x = Float64[]
y = Float64[]
z = Float64[]
for u in us, v in vs
    push!(x, u); push!(y, u); push!(z, v)   # plane x=y
end

# 三角形フェイスを作成
i = Int[]
j = Int[]
k = Int[]
for ii in 1:m-1
    for jj in 1:n-1
        p1 = (ii-1)*n + jj
        p2 = ii*n       + jj
        p3 = (ii-1)*n + jj+1
        p4 = ii*n       + jj+1
        # 2 つの三角形に分割
        push!(i, p1-1); push!(j, p2-1); push!(k, p3-1)
        push!(i, p2-1); push!(j, p4-1); push!(k, p3-1)
    end
end

trace = mesh3d(
    x=x, y=y, z=z,
    i=i, j=j, k=k,                # ★自前の分割インデックス
    opacity=0.5, color="red",
    showscale=false
)
